:root {
  $names: map.get(cfg.$merged, common, sizes);
  $func: map.get(cfg.$merged, layout, container-function);

  @each $name, $value in $names {
    $sizes: map.set($sizes, $name, meta.call($func, $value)) !global;
    --container-#{$name}: #{map.get($sizes, $name)};
  }

Line Heights

scss 
:root {
  $scale-base: map.get(cfg.$merged, typography, scale, base);
  $scale-factor: map.get(cfg.$merged, typography, scale, factor);
  $scale-min: map.get(cfg.$merged, typography, scale, min);
  $scale-max: map.get(cfg.$merged, typography, scale, max);

  $line-min: map.get(cfg.$merged, typography, line, min);
  $line-root: map.get(cfg.$merged, typography, line, root);

  $rlh: $line-root * $scale-base !global;

  --line-root: #{$line-root};
  --line-root-abs: #{$rlh};

  @for $i from $scale-min through $scale-max {
    $font-size: scale.modular($scale-base, $scale-factor, $i);
    --line-#{$i}: #{rhythm.grid-line($font-size, $rlh, $line-min)};
  }
}

Line heights supporting the vertical rhythm at every step of the scale. Defines $rlh.

Modular Font Size Scale

scss 
:root {
  $base: map.get(cfg.$merged, typography, scale, base);
  $factor: map.get(cfg.$merged, typography, scale, factor);
  $min: map.get(cfg.$merged, typography, scale, min);
  $max: map.get(cfg.$merged, typography, scale, max);

  --scale-factor: #{$factor};

  @for $i from $min through $max {
    --scale-#{$i}: #{scale.modular($base, $factor, $i)};
  }

  --scale-smaller: #{math.div(100%, $factor)};
  --scale-larger: #{100% * $factor};
}

Stroke Width Scale

scss 
:root {
  $min: map.get(cfg.$merged, ui, stroke, min);
  $max: map.get(cfg.$merged, ui, stroke, max);
  $line: map.get(cfg.$merged, ui, stroke, line);

  @for $i from $min through $max {
    --stroke-width-#{$i}: #{line.$rlh * math.div($i, $line)};
  }
}

A scale of stroke widths with $line units per root line.
